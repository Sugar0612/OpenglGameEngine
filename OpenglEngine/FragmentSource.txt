#version 330 core	

in vec4 color;                          
in vec2 TexCoord;

uniform vec4 vertexColor;   

uniform sampler2D aTexture;
uniform sampler2D aFace;

out vec4 FragColor;	
void main()								
{					
    FragColor = mix(texture(aTexture, TexCoord), texture(aFace, TexCoord), 0.2);				
}
